## Cоздаем таблицы
```SQL
CREATE TABLE IF NOT EXISTS author(
	author_id BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
	name VARCHAR(150) NOT NULL CHECK(LENGTH(name) >= 3),
	description TEXT CHECK(LENGTH(description) >= 30));

CREATE TABLE IF NOT EXISTS book_cathegory(
	cathegory_id INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
	name VARCHAR(150) NOT NULL CHECK(LENGTH(name) >= 3));
	
CREATE TABLE IF NOT EXISTS book(
	book_id BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
	cathegory_id INT NOT NULL REFERENCES book_cathegory(cathegory_id),
	author_id BIGINT NOT NULL REFERENCES author(author_id),
	name VARCHAR(150) NOT NULL CHECK(LENGTH(name) >= 3),
	description TEXT CHECK(LENGTH(description) >= 30),
	cover VARCHAR(255));
```
